1function [u_out_P1, broad_interv_P1, u_out_P2]  = Tossim_block(u_in_P1, u_in_P2, clock)
  2% Inputs:
  3% ymea: measurement from structure
  4% Outputs:
  5% ymea1: measurement output
  6% ydelay: delay traces
  7persistent tmp_u_in_P1 tmp_u_in_P2 prev_u_out_P1 prev_u_out_P2    % delay trace
  8persistent prev_P1_actu_inter prev_P2_actu_inter prev_P1_clock prev_P2_clock % inter arrival time
  9persistent tmp_T1_period tmp_T2_period tmp_rcv_flag1 period_cunt1 windowbuff windowsize
 10persistent tmp_T1_maxTx tmp_T1_noise
 11persistent i                 % count 
 12persistent tossim_flag
 13global ydelay_global delta Task_periods
 14
 15
 16% allow these matlab functions to be used in Embedded Matlab block
 17eml.extrinsic('python');
 18eml.extrinsic('num2str');
 19eml.extrinsic('str2num');
 20eml.extrinsic('delay_summary');
 21eml.extrinsic('disp');
 22eml.extrinsic('ps_aggregation_broadcast_interval');
 23eml.extrinsic('for');
 24eml.extrinsic('end');
 25
 26if isempty(i)
 27    i = 0;
 28    %past_ymea = 0;
 29    tmp_u_in_P1 = u_in_P1;
 30    tmp_u_in_P2 = u_in_P2;
 31    prev_u_out_P1 = 0;
 32    prev_u_out_P2 = 0;
 33    tossim_flag = 0;
 34    prev_P1_actu_inter = 0;
 35    prev_P2_actu_inter = 0;
 36    prev_P1_clock = 0;
 37    prev_P2_clock = 0;
 38    P1_actu_inter = 0;
 39    P2_actu_inter = 0;
 40    tmp_T1_period = 0;
 41    tmp_T2_period = 0;
 42    tmp_rcv_flag1 = 0;
 43    period_cunt1 = 0;
 44    windowsize = 20;
 45    windowbuff = zeros(1,windowsize);
 46    tmp_T1_maxTx = 0;
 47    tmp_T1_noise = 0;
 48end
 49i = i+1;
 50
 51%[row column] = size(Task_period_time);
 52
 53ydelay = [1 0 0 0 0 0 0 0 0]';
 54% call python to communicate with TOSSIM
 55
 56% where clock increase with 1ms period
 57TOSSIM_step_size = 0.01;  % 10ms
 58
 59call_count = 1; %Superframelength
 60rssi = -69;%sensor rssi
 61
 62%% Tossim results 
 63% run Tossim once every 10ms
 64if mod(clock, TOSSIM_step_size) == 0
 65    if tossim_flag == 0
 66        tossim_flag = tossim_flag + 1;
 67        clock;
 68        [resultz] = ps_aggregation_broadcast_interval(call_count, rssi, clock); %here call TOSSIM Client ----yh
 69        end_to_end_delivery_result_aggregated = sum(resultz);
 70        [delay] = delay_summary(resultz);
 71        ydelay = delay';
 72        resultz;
 73        
 74        % Check T1 Period %
 75        tmp_T1_period = resultz(1,8) / 100; % convert into ms
 76        % Check T1 maxTx %
 77        tmp_T1_maxTx = resultz(1,7);
 78        % Check noise %
 79        tmp_T1_noise = resultz(1,6);
 80        
 81        %% Check P1 actuation PDR
 82        % Check T1 receive in a period %   
 83        if ydelay(2,1) == 1     % Task1 receive
 84            tmp_rcv_flag1 = 1;
 85        end
 86        
 87        % for calculating PDR and for updating u_in_P1
 88        if mod(clock, tmp_T1_period) == 0 
 89            % PDR 
 90            tmp_period_cunt1 = mod(period_cunt1, windowsize);
 91            if tmp_rcv_flag1 == 1
 92                windowbuff(1,tmp_period_cunt1+1) = 1;
 93            else
 94                windowbuff(1,tmp_period_cunt1+1) = 0;
 95            end
 96            period_cunt1 = period_cunt1 + 1;
 97            tmp_rcv_flag1 = 0;
 98            
 99            % update
100            tmp_u_in_P1 = u_in_P1;
101        end
102        
103        % P2 u update
104        tmp_T2_period = Task_periods(2,1)*0.01;
105        if mod(clock, tmp_T2_period) == 0
106            tmp_u_in_P2 = u_in_P2;
107        end
108
109        
110    else
111        resultz = [0 0 0 0 0 0 0 0];
112        [delay] = delay_summary(resultz);
113        ydelay = delay';
114    end
115else
116    tossim_flag = 0;
117    resultz = [0 0 0 0 0 0 0 0];
118    [delay] = delay_summary(resultz);
119    ydelay = delay';
120end
121
122
123
124%% Outputs
125for i_yg = 1:9
126    ydelay_global(i_yg, 1) = ydelay(i_yg, 1);
127end
128delay = ydelay';
129
130task1_index = 1;
131task2_index = 2;
132% ?? Can't multiple packets reach the desitnation at the same time??
133if delay(1,task1_index + 1)~=88
134    u_out_P1 = tmp_u_in_P1;
135    prev_u_out_P1 = u_out_P1;
136    u_out_P2 = prev_u_out_P2;
137    %%%%
138    % check application interval
139    P1_actu_inter = clock - prev_P1_clock;
140    prev_P1_clock = clock;
141    prev_P1_actu_inter = P1_actu_inter;
142    
143    if is_changed(tmp_T1_period)
144        broad_interv_P1 = compute_broadcast_interval(P1_actu_inter, clock);
145    else
146        broad_interv_P1 = 0;
147    end
148    
149    P2_actu_inter = prev_P2_actu_inter;
150elseif delay(1,task2_index + 1)~=88
151    u_out_P2 = tmp_u_in_P2;
152    prev_u_out_P2 = u_out_P2;
153    u_out_P1 = prev_u_out_P1;
154    %%%
155    P2_actu_inter = clock - prev_P2_clock;
156    prev_P2_clock = clock;
157    prev_P2_actu_inter = P2_actu_inter;
158    P1_actu_inter = prev_P1_actu_inter;
159else
160    u_out_P1 = prev_u_out_P1;
161    u_out_P2 = prev_u_out_P2;
162    P1_actu_inter = prev_P1_actu_inter;
163    P2_actu_inter = prev_P2_actu_inter;
164end
165
166T1_period = tmp_T1_period;
167P1_PDR = sum(windowbuff)/windowsize;
168T1_maxTx = tmp_T1_maxTx;
169T1_noise = tmp_T1_noise;
170broad_interv_P1 = 0
171end

176function [broad_interv] = compute_broadcast_interval(actu_interv, curr_time)
177global Dis_instant_P1
178dist_inst = Dis_instant_P1
179
180if actu_interv == changed_T1_period

184end

186end

188% check if current task period is changed or not
189function [output] = is_changed(curr_task_period)
190persistent prev_T1_period
191
192if isempty(prev_T1_period)
193    prev_T1_period = curr_task_period;
194    output = False;
195    return
196end
197
198if prev_T1_period == curr_task_period;
199    output = True;
200else
201    output = False;
202
203end
